2020/09/25

一、基本系统安装

修改BIOS设置

关闭安全启动

设置Windows

关闭Windows快速启动

设置tty字体

setfont ter-132n

首次连接网络

iwctl

device list

station wlan0 scan

station wlan0 get-networks

station wlan0 connect SSID

quit

设置系统时间

timedatectl set-ntp true

格式化并挂载分区

lsblk

格式化根分区：

mkfs.btrfs -L Arch /dev/sdaX

挂载根分区：

mount /dev/sdaX /mnt

创建子卷：

btrfs su cr /mnt/@

btrfs su cr /mnt/@home

卸载根分区：

umount /mnt

挂载子卷：

mount -o noatime,discard,ssd,space_cache,compress=lzo,subvol=@ /dev/sdaX /mnt

mkdir /mnt/home

mount -o noatime,discard,ssd,space_cache,compress=lzo,subvol=@home /dev/sdaX /mnt/home

挂载ESP分区：

mkdir /mnt/boot

monut /dev/sdaY /mnt/boot

安装系统基础软件包

pacstrap /mnt base base-devel linux-lts linux-lts-headers linux-firmware btrfs-progs dosfstools ntfs-3g intel-ucode grub efibootmgr os-prober sudo fish iwd vim man-db man-pages

生成fstab

genfstab -U /mnt >> /mnt/etc/fstab

cat /mnt/etc/fstab

进入新系统

arch-chroot /mnt

时区

ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime

hwclock --systohc

本地化

vim /etc/locale.gen

en_US.UTF-8 UTF-8
zh_CN.UTF-8 UTF-8
ja_JP.UTF-8 UTF-8

locale-gen

echo LANG=en_US.UTF-8 > /etc/locale.conf

主机名

echo matebook > /etc/hostname

hosts

vim /etc/hosts

127.0.0.1	localhost
::1	localhost
127.0.0.1	matebook.localdomain	matebook

配置initramfs参数

vim /etc/mkinitcpio.conf

HOOKS=(add btrfs)

mkinitcpio -p linux

设置root密码

passwd

安装引导程序

grub-install --target=x86_64-efi --efi-directory=/boot --bootloader-id=Arch

grub-mkconfig -o /boot/grub/grub.cfg

重启

exit

reboot

重新连接网络

方案一：

systemctl enable --now systemd-networkd.service

systemctl enable --now systemd-resolved.service

systemctl enable --now iwd.service

vim /etc/iwd/main.conf

[General]
EnableNetworkConfiguration=true
[Network]
NameResolvingService=systemd

systemctl restart iwd.service

iwctl

方案二：

vim /etc/systemd/network/config.network

[Match]
Name=interface
[Network]
DHCP=yes

systemctl enable --now systemd-networkd.service

ln -s /run/systemd/resolve/stub-resolv.conf /etc/resolv.conf

systemctl enable --now systemd-resolved.service

systemctl enable --now iwd.service

添加新用户

useradd -m -G wheel huizhi

passwd huizhi

vim /etc/sudoers

去掉# %wheel ALL=(ALL)ALL 之前的注释符#

四、完善配置

自动挂载windows分区

sudo mkdir /media

sudo mkdir /media/Windows

sudo mkdir /media/Data

sudo vim /etc/fstab

/dev/sda4       /media/Windows    ntfs-3g         uid=huizhi,gid=users,dmask=022,fmask=133    0       0
/dev/sdb2       /media/Data    ntfs-3g         uid=huizhi,gid=users,dmask=022,fmask=133    0       0

建立常用目录的软链接

ln -s /media/Data/Hugo ~/Hugo

rm -r ~/Movies && ln -s /media/Data/Videos ~/Videos

rm -r ~/Music && ln -s /media/Data/Music ~/Music

rm -r ~/Pictures && ln -s /media/Data/Pictures ~/Pictures

rm -r ~/Documents && ln -s /media/Data/Documents ~/Documents

从Windows安装字体

sudo mkdir /usr/share/fonts/WindowsFonts

sudo cp /media/Windows/Windows/Fonts/* /usr/share/fonts/WindowsFonts

sudo chmod 755 /usr/share/fonts/WindowsFonts/*

fc-cache -f

恢复配置文件

git clone https://github.com/qvshuo/dotfiles.git

xprop | grep WM_CLASS

使Windows使用UTC时间

Windows下以管理员权限运行：

reg add "HKEY_LOCAL_MACHINE\System\CurrentControlSet\Control\TimeZoneInformation" /v RealTimeIsUniversal /d 1 /t REG_QWORD /f
